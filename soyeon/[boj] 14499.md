## [주사위 굴리기]()

```java
import java.util.*;

public class Main {
	public static void main(String args[]) {
		Scanner in = new Scanner(System.in);
		// 지도 가로 세로
		int N = in.nextInt();
		int M = in.nextInt();
		// 주사위 위치 (row, col)
		int x = in.nextInt();
		int y = in.nextInt();
		Point point = new Point(x, y);
		// 명령 개수
		int K = in.nextInt();

		int map[][] = new int[N][M];
		for (int r = 0; r < N; r++) {
			for (int c = 0; c < M; c++) {
				map[r][c] = in.nextInt();
			}
		}

		int dice[] = new int[7];
		for (int i = 0; i < K; i++) {
			int move = in.nextInt();
			if (moveDice(dice, map, point, move, N, M)) {
				System.out.println(dice[1]);
			}
		}
	}

	public static boolean moveDice(int dice[], int map[][], Point point, int move, int N, int M) {
		if (move == 1) {// 동
			if (inBoundary(point.row, point.col + 1, N, M)) {
				point.col += 1;

				int temp = dice[1];
				dice[1] = dice[3];
				dice[3] = dice[6];
				dice[6] = dice[4];
				dice[4] = temp;

				if (map[point.row][point.col] == 0) {
					map[point.row][point.col] = dice[6];
				} else {
					dice[6] = map[point.row][point.col];
					map[point.row][point.col] = 0;
				}
				return true;
			}
		} else if (move == 2) {// 서
			if (inBoundary(point.row, point.col - 1, N, M)) {
				point.col -= 1;

				int temp = dice[1];
				dice[1] = dice[4];
				dice[4] = dice[6];
				dice[6] = dice[3];
				dice[3] = temp;

				if (map[point.row][point.col] == 0) {
					map[point.row][point.col] = dice[6];
				} else {
					dice[6] = map[point.row][point.col];
					map[point.row][point.col] = 0;
				}
				return true;
			}
		} else if (move == 3) {// 북
			if (inBoundary(point.row - 1, point.col, N, M)) {
				point.row -= 1;

				int temp = dice[1];
				dice[1] = dice[2];
				dice[2] = dice[6];
				dice[6] = dice[5];
				dice[5] = temp;

				if (map[point.row][point.col] == 0) {
					map[point.row][point.col] = dice[6];
				} else {
					dice[6] = map[point.row][point.col];
					map[point.row][point.col] = 0;
				}
				return true;
			}
		} else if (move == 4) {// 남
			if (inBoundary(point.row + 1, point.col, N, M)) {
				point.row += 1;

				int temp = dice[1];
				dice[1] = dice[5];
				dice[5] = dice[6];
				dice[6] = dice[2];
				dice[2] = temp;

				if (map[point.row][point.col] == 0) {
					map[point.row][point.col] = dice[6];
				} else {
					dice[6] = map[point.row][point.col];
					map[point.row][point.col] = 0;
				}
				return true;
			}
		}
		return false;
	}

	public static boolean inBoundary(int row, int col, int N, int M) {
		if (row < 0) {
			return false;
		}
		if (row >= N) {
			return false;
		}
		if (col < 0) {
			return false;
		}
		if (col >= M) {
			return false;
		}
		return true;
	}
}

class Point {
	int row, col;

	public Point(int row, int col) {
		this.row = row;
		this.col = col;
	}
}
```
